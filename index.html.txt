<script>

const WIDTH = 0.915
const LENGTH = 1.83

document.querySelectorAll("input").forEach(i=>{
i.addEventListener("input",calc)
i.addEventListener("blur",formatInput)
})

function number(id){
return parseFloat(
document.getElementById(id).value.replace(/,/g,'')
)||0
}

function format(x){
return Number(x).toLocaleString("en-US",{maximumFractionDigits:2})
}

function formatInput(e){

let v=e.target.value.replace(/,/g,'')

if(v!==""){
e.target.value=Number(v).toLocaleString("en-US")
}

}

function calc(){

let thickness=number("thickness")
let price=number("price")
let bundle=number("bundle")
let pcsBundle=number("pcsBundle")
let usd=number("usd")
let sellpc=number("sellpc")

let pcs=bundle*pcsBundle

let cbmPc=(thickness/1000)*WIDTH*LENGTH

let cbmTotal=pcs*cbmPc

let buyUsd=pcs*price

let buyVnd=buyUsd*usd

let sellVnd=pcs*sellpc

let profitTotal=Math.round(sellVnd-buyVnd)

let profitPc=pcs?profitTotal/pcs:0

document.getElementById("pcs").innerText=format(pcs)

document.getElementById("cbmPc").innerText=format(cbmPc)

document.getElementById("cbmTotal").innerText=format(cbmTotal)

document.getElementById("buyUsd").innerText=format(buyUsd)

document.getElementById("buyVnd").innerText=format(buyVnd)

document.getElementById("sellVnd").innerText=format(sellVnd)

let ppc=document.getElementById("profitPc")
ppc.innerText=format(profitPc)

let pt=document.getElementById("profitTotal")
pt.innerText=format(profitTotal)

if(profitTotal>=0){
ppc.className="blue"
pt.className="blue"
}else{
ppc.className="red"
pt.className="red"
}

}

if("serviceWorker" in navigator){
navigator.serviceWorker.register("service-worker.js")
}
